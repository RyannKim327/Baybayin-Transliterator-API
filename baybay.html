 This is my simple roman to baybayin transliterator api powered by Vercel App and ExpressJS.

Sample code:

					
const axios = require("axios")

async function transliterate(text){
	let result = await axios.get("https://api-baybayin-transliterator.vercel.app/?text=" + text).then((r) => {  
		return r.data
	}).catch((e) => {
		console.error("Error [Baybayin API]: " + e)
		return null
	})
	return result
}

module.exports = async (data) => {

	let result = await transliterate(data)
	console.log(result)
